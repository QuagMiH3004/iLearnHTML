css
Đặt style ở head
color // màu chữ
background-color //  màu nền
font-size: cỡ chữ (vd 20px)
width / height: rộng/ cao // tính bằng px hoặc % (tỉ lệ màn hình)
border // đường viền (border: [độ dày] [kiểu viền] [màu sắc])
        1px solid red
    kiểu viền: solid: Đường kẻ liền (Phổ biến nhất).
                dashed: kẻ đứt quãng (---).
                dotted: kẻ chấm bi (......).
                double: kẻ đôi.
                none: Không có viền (xóa viền của button hoặc input).
text-align: căn lề chữ (left/ right/ top/ bottom)
class ký hiệu bằng dấu . (ở style trên css)
id ký hiệu bằng dấu #
// viết liên tiếp nhiều thẻ cần 1 thuộc tính bằng dấu "," h1, p, li, ul

Box model // bố cục web
 - Content (nội dung) là chữ hoặc hình ảnh (cốt lõi)
 - Padding // khoảng cách từ nội dung đến viền
 - Border // viền ngoài bao quanh
 - Margin // khoảng cách từ viền này đến viền khác
:hover // dùng cho button khi di chuột vào
:active // dùng cho button khi ấn xuống
font-weight để chỉnh font chữ đậm nhạt
display // dùng để quyết định 1 thẻ được hiển thị thế nào
        - display: block // khối, chiếm toàn bộ chiều ngang và buộc thẻ sau xuống hàng. Chỉnh được width và height.
        - display: inline // chiểm vừa khít không gian cho nội dung trong nó, 2 thẻ inline sẽ xếp cạnh nhau.
        - display: inline-block // xếp hàng ngang với thẻ khác trên cùng 1 dòng nhưng chỉnh được width/ heigt.
        - display: none // ẩn hoàn toàn khối đó đi
transition: [thuộc tính] [thời gian] [kiểu chuyển động]
    - all// áp dụng kiểu chuyển động cho tất cả mọi sự thay đổi. Nếu muốn thay đổi riêng thì viết riêng (vd: background-color 0.2s)
        - thời gian tính bằng s (0.2s) hoặc ms (200ms).
            - Kiểu chuyển động: ease // chậm - tăng tốc - chậm
                                linear // liên tục không nhanh chậm
                                ease-in // chậm - nhanh dần lên
                                ease-out // nhanh- chậm dần lại
cursor: con chuột, chỉnh xem phản ứng thế nào khi chuột di vào
        - pointer // con trỏ (bàn tay chỉ vào, ấn vào được)
        - not-allowed // biến thành hình cấm, gạch chéo
        - wait // đồng hồ cá loading
        - text // hình chữ I
        - zoom-in / zoom-out // hình kính lúp, thường dùng cho ảnh
transform: thay đổi hình dáng, kích thước
        - scale (0.98) // thu nhỏ đi còn 98% kích thước gốc vật tể
        - rotate(45deg) // quay 45 độ
        - translateX(20px) // Dịch chuyển vật thể sang bên phải 20 pixel.
margin : khoảng cách [trên] [phải] [dưới] [trái]

        grayscale(100%) // biến phía sau thành đen trắng
